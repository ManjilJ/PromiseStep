const getUser = id => new Promise(resolve => {
const greeting = "Welcome back,";
const fetchUserName = cb => setTimeout(() => cb("Alice"), 1000);
fetchUserName(name => {
console.log(greeting + " " + name); // Closure: captures 'greeting'
});
console.log("Request sent, waiting for response...");
setTimeout(() => resolve({id, name: 'Alice'}), 1500);
});
const getPosts = id => new Promise(res => setTimeout(() => res([{id: 101}]), 1000));
const getComments = id => new Promise(res => setTimeout(() => res([{text: 'Comment A'}]), 1000));
getUser(1)
.then(user => getPosts(user.id))
.then(posts => getComments(posts[0].id))
.then(data => console.log("Final:", data[0].text));
Ladder
getUser(1)
.then(getPosts)
.then(getComments)
Final Output
Call Stack (synchronous, LIFO)
Web APIs (timers in flight)
Macrotask Queue (setTimeout callbacks, FIFO)
Microtask Queue (.then continuations — always drained first)
